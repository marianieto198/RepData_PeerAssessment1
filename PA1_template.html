###2.Devise a strategy for filling in all of the missing values in the dataset. ####Replacing NAs with the mean of the 5-minute interval

for (i in 1:nrow(Data1)) {
    if(is.na(Data1$steps[i])) {
      mis <- steps_interval$steps[which(steps_interval$interval == Data1$interval[i])]
        Data1$steps[i] <- mis
    }
}
